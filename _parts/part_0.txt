function uid() { return Date.now().toString(36) + Math.random().toString(36).slice(2,7); }
function favicon(url) { try { return 'https://www.google.com/s2/favicons?domain='+new URL(url).hostname+'&sz=32'; } catch{return '';} }
function fuzzy(q, t) { q=q.toLowerCase(); t=t.toLowerCase(); if(t.includes(q)) return true; let qi=0; for(let i=0;i<t.length&&qi<q.length;i++) if(t[i]===q[qi]) qi++; return qi===q.length; }
function getGreeting() { const h=new Date().getHours(); if(h<6) return 'Good night'; if(h<12) return 'Good morning'; if(h<18) return 'Good afternoon'; return 'Good evening'; }
function isMac() { return navigator.platform.toUpperCase().indexOf('MAC')>=0; }
function hexToHsl(hex) {
  let r=parseInt(hex.slice(1,3),16)/255, g=parseInt(hex.slice(3,5),16)/255, b=parseInt(hex.slice(5,7),16)/255;
  const max=Math.max(r,g,b), min=Math.min(r,g,b), l=(max+min)/2;
  let h=0, s=0;
  if(max!==min){ const d=max-min; s=l>0.5?d/(2-max-min):d/(max+min);
    if(max===r) h=((g-b)/d+(g<b?6:0))/6; else if(max===g) h=((b-r)/d+2)/6; else h=((r-g)/d+4)/6; }
  return [Math.round(h*360), Math.round(s*100), Math.round(l*100)];
}
function hslToHex(h,s,l) {
  s/=100; l/=100;
  const a=s*Math.min(l,1-l);
  const f=n=>{ const k=(n+h/30)%12; return l-a*Math.max(Math.min(k-3,9-k,1),-1); };
  return '#'+[f(0),f(8),f(4)].map(x=>Math.round(x*255).toString(16).padStart(2,'0')).join('');
}
function luminance(hex) {
  const r=parseInt(hex.slice(1,3),16)/255, g=parseInt(hex.slice(3,5),16)/255, b=parseInt(hex.slice(5,7),16)/255;
  const toLinear = c => c<=0.03928 ? c/12.92 : Math.pow((c+0.055)/1.055, 2.4);
  return 0.2126*toLinear(r) + 0.7152*toLinear(g) + 0.0722*toLinear(b);
}
function deriveColorsFromBg(bgHex, contrast = 50) {
  const lum = luminance(bgHex);
  const isLight = lum > 0.35;
  const cf = 0.5 + (contrast / 100);
  if (isLight) {
    const primaryL = Math.round(7 * (2 - cf));
    const secondaryL = Math.round(33 * (2 - cf));
    const tertiaryL = Math.round(60 * (2 - cf));
    return {
      '--surface': bgHex,
      '--surface-secondary': darkenHex(bgHex, Math.round(4 * cf)),
      '--surface-hover': darkenHex(bgHex, Math.round(8 * cf)),
      '--text-primary': isLight ? hslToHex(0, 0, Math.min(primaryL, 30)) : '#e8e8e8',
      '--text-secondary': isLight ? hslToHex(0, 0, Math.min(secondaryL, 65)) : hslToHex(0, 0, Math.max(40, secondaryL)),
      '--text-tertiary': isLight ? hslToHex(0, 0, Math.min(tertiaryL, 85)) : hslToHex(0, 0, Math.max(25, tertiaryL)),
      '--border': darkenHex(bgHex, Math.round(10 * cf)),
    };
  } else {
    const primaryL = Math.round(91 - 7 * (2 - cf));
    const secondaryL = Math.round(60 - 9 * (2 - cf));
    const tertiaryL = Math.round(40 - 10 * (2 - cf));
    return {
      '--surface': bgHex,
      '--surface-secondary': lightenHex(bgHex, Math.round(5 * cf)),
      '--surface-hover': lightenHex(bgHex, Math.round(10 * cf)),
      '--text-primary': hslToHex(0, 0, primaryL),
      '--text-secondary': hslToHex(0, 0, secondaryL),
      '--text-tertiary': hslToHex(0, 0, tertiaryL),
      '--border': lightenHex(bgHex, Math.round(12 * cf)),
    };
  }
}
function darkenHex(hex, amount) { const [h,s,l] = hexToHsl(hex); return hslToHex(h, s, Math.max(0, l - amount)); }
function lightenHex(hex, amount) { const [h,s,l] = hexToHsl(hex); return hslToHex(h, s, Math.min(100, l + amount)); }

// ========== THEME ==========
function applyTheme(theme) {
  if(theme==='dark') document.body.className='dark';
  else if(theme==='light') document.body.className='';
  else document.body.className = window.matchMedia('(prefers-color-scheme:dark)').matches?'dark':'';
  const icon = document.getElementById('themeIcon');
  const isDark = document.body.classList.contains('dark');
  icon.innerHTML = isDark ? ICONS.sun : ICONS.moon;
  applyBackground();
}
function applyBackground() {
  const bg = state.settings.background;
  let effectiveBg = '#1a1a1a';
  if(bg.type==='solid') {
    effectiveBg = bg.value;
    document.body.style.background = bg.value;
    document.body.style.backgroundImage = '';
  } else if(bg.type==='gradient') {
    const p = bg.value.split(',');
    effectiveBg = p[0] || '#1a1a1a';
    document.body.style.background = `linear-gradient(${p[2]||135}deg, ${p[0]||'#1a1a1a'}, ${p[1]||'#2a2a2a'})`;
    document.body.style.backgroundImage = '';
  } else if(bg.type==='api') {
    effectiveBg = '#111111';
    loadApiImage(bg.value);
    return;
  } else {
    const id = bg.value || '1065976';
    effectiveBg = '#1a1a1a';
    document.body.style.background = '';
    document.body.style.backgroundImage = `url(https://source.unsplash.com/collection/${id}/1920x1080/daily)`;
    document.body.style.backgroundSize = 'cover';
    document.body.style.backgroundPosition = 'center';
  }
  const colors = deriveColorsFromBg(effectiveBg, state.settings.contrast || 50);
  const root = document.documentElement;
  Object.entries(colors).forEach(([prop, val]) => root.style.setProperty(prop, val));
}

const API_IMAGE_SOURCES = {
  beauty: { name: '美女', url: 'https://api.eyabc.cn/api/picture/beauty' },
  anime: { name: '动漫', url: 'https://api.eyabc.cn/api/picture/dong_man' },
  landscape: { name: '风景', url: 'https://api.eyabc.cn/api/picture/scenery' },
};
function loadApiImage(category) {
  const source = API_IMAGE_SOURCES[category];
  if(!source) return;
  const apiUrl = source.url + '?r=' + Math.random().toString(36).slice(2);
  document.body.style.background = '#1a1a1a';
  document.body.style.backgroundImage = '';
  fetch(apiUrl, { mode: 'cors', redirect: 'follow' })
    .then(response => { if(!response.ok) throw new Error('API error'); return response.blob(); })
    .then(blob => {
      const url = URL.createObjectURL(blob);
      document.body.style.background = '';
      document.body.style.backgroundImage = `url(${url})`;
      document.body.style.backgroundSize = 'cover';
      document.body.style.backgroundPosition = 'center';
      document.body.style.backgroundRepeat = 'no-repeat';
      document.body.style.backgroundAttachment = 'fixed';
      const colors = deriveColorsFromBg('#111111', state.settings.contrast || 50);
      const root = document.documentElement;
      Object.entries(colors).forEach(([prop, val]) => root.style.setProperty(prop, val));
    })
    .catch(() => { document.body.style.background = '#1a1a1a'; document.body.style.backgroundImage = ''; });
}
